 26def pipe(f, obj, *args, at = None, **kwargs):
 27    if at is None:
 28        return f(obj, *args, **kwargs)
 29    elif isinstance(at, int):
 30        return f(*args[:at], obj, *args[at:], **kwargs)
 31    elif isinstance(at, str):
 32        return f(*args, **{at: obj}, **kwargs)
 33    else:
 34        assert False, at
